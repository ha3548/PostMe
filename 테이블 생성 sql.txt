/* PROFILE TABLE */

CREATE TABLE PROFILE(
   ID VARCHAR2(25) CONSTRAINT PK_PRO_ID PRIMARY KEY,
    CONSTRAINT FK_PRO_ID FOREIGN KEY(ID) REFERENCES USERS(ID),
   NAME VARCHAR2(30),
   AGE NUMBER(2),
   PHONE CHAR(13),
   GENDER CHAR(6),
   INFORMATION VARCHAR2(100),
   PROFILE_IMAGE VARCHAR2(100)
);
